    private fun registerWifiChangeReceiver() {
        val filter = android.content.IntentFilter(WifiManager.NETWORK_STATE_CHANGED_ACTION)
        wifiChangeReceiver = object : android.content.BroadcastReceiver() {
            private var lastSsid: String? = null
            override fun onReceive(context: Context, intent: Intent) {
                val info = if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.Q) {
                    null // use ConnectivityManager on Q+
                } else {
                    @Suppress("DEPRECATION")
                    (intent.getParcelableExtra(WifiManager.EXTRA_NETWORK_INFO) as? android.net.NetworkInfo)
                }
                val connected = info?.isConnected ?: run {
                    val cm = getSystemService(Context.CONNECTIVITY_SERVICE) as ConnectivityManager
                    cm.getNetworkCapabilities(cm.activeNetwork)
                        ?.hasCapability(android.net.NetworkCapabilities.NET_CAPABILITY_INTERNET) == true
                }
                if (connected) {
                    val wm   = applicationContext.getSystemService(Context.WIFI_SERVICE) as WifiManager
                    @Suppress("DEPRECATION")
                    val ssid = wm.connectionInfo?.ssid?.removeSurrounding("\"")
                    if (ssid != null && ssid != lastSsid && ssid != "<unknown ssid>") {
                        lastSsid = ssid
                        val prefs = getSharedPreferences(PREFS, Context.MODE_PRIVATE)
                        val alreadySaved = prefs.contains("$PREF_WIFI_PASS_PFX$ssid")
                        if (!alreadySaved) showWifiPasswordPrompt(ssid)
                    }
                }
            }
        }
        registerReceiver(wifiChangeReceiver, filter)
    }

    class ImageAssembler(private val total: Int) {
        private val received = BooleanArray(total)
        private val buffers = arrayOfNulls<ByteArray>(total)
        private var receivedCount = 0

        fun insert(seq: Int, data: ByteArray) {
            if (seq < 0 || seq >= total) return
            if (received[seq]) return

            buffers[seq] = data
            received[seq] = true
            receivedCount++
        }

        fun isComplete(): Boolean = receivedCount == total

        fun build(): ByteArray {
            val size = buffers.sumOf { it?.size ?: 0 }
            val out = ByteArray(size)
            var pos = 0
            for (i in 0 until total) {
                val b = buffers[i] ?: continue
                System.arraycopy(b, 0, out, pos, b.size)
                pos += b.size
            }
            return out
        }
    }

    class MemoryRetrieveService : Service() {
        private var worker: ForegroundWorker? = null

        override fun onCreate() {
            super.onCreate()
            val mainActivity = activeInstance ?: run {
                android.util.Log.e("SERVICE", "activeInstance is null — MainActivity not ready")
                return
            }
            worker = mainActivity.ForegroundWorker(this)
            worker?.startForegroundCompat(this)
            worker?.start()
            // hand reference back to MainActivity
            mainActivity.worker = worker
        }

        override fun onDestroy() {
            super.onDestroy()
            worker?.stopWorker()
            worker = null
        }

        override fun onBind(intent: Intent?): IBinder? = null
    }

    private val uiRefresh = object : Runnable {
        override fun run() {
            try {
                worker?.let { w ->
                    // update live image if exists
                    w.latestVideoFrame?.let { bmp ->
                        imgLive.setImageBitmap(bmp)
                    }

                    // copy detections from worker
                    synchronized(detections) {
                        detections.clear()
                        detections.addAll(w.detections.map { det ->
                            Detection(det.filename, det.className, det.trackId, det.ts, det.lat, det.lon, det.rawBytes, det.bitmap, Collections.synchronizedList(det.boxes.toMutableList()))
                        })
                    }
                    val mode = determineSystemMode()
                    val nodeCount = nodeRegistry.size
                    tvStatus.text = "PI: ${w.piIp ?: "—"}  |  NODES: $nodeCount  |  MODE: $mode"
                    composeNodeCount.intValue = nodeCount
                    composeMode.value = mode.name
                    composeStatus.value = "PI: ${w.piIp ?: "—"}  |  NODES: $nodeCount"
                }
                filterDetections()

                // update orchestration UI (routes)
                updateRoutePanel()

            } catch (e: Exception) {
                Log.e("MA-UI", "refresh error: ${e.message}")
            } finally {
                handler.postDelayed(this, 800)
            }
        }
    }

    private fun filterDetections() {
        val query = etSearch.text.toString().lowercase()
        filtered.clear()
        filtered.addAll(if (query.isBlank()) detections else detections.filter { it.className.lowercase().contains(query) })
        (grid.adapter as BaseAdapter).notifyDataSetChanged()
    }

    // route computing & UI
    data class RouteOption(val key: String, val label: String, val score: Int)

    private fun computeRoutes(nodes: List<NodeInfo>): List<RouteOption> {
        val sorted = nodes.sortedByDescending { computeNodeScore(it) }

        val roles = sorted.map { it.role.uppercase() }.toSet()

        val hasZero = roles.contains("PI_ZERO")
        val hasPi5 = roles.contains("PI5")
        val hasLaptop = roles.contains("LAPTOP")

        val list = mutableListOf<RouteOption>()

        if (hasZero && hasLaptop) list.add(RouteOption("Z_P_L", "PI ZERO + PHONE + LAPTOP", 100))
        if (hasLaptop && hasPi5) list.add(RouteOption("P_5_L", "PHONE + PI5 + LAPTOP", 90))
        if (hasZero && hasPi5) list.add(RouteOption("Z_P_5", "PI ZERO + PHONE + PI5", 85))
        if (hasZero) list.add(RouteOption("Z_P", "PI ZERO + PHONE", 70))
        if (hasPi5) list.add(RouteOption("P_5", "PHONE + PI5", 65))

        if (list.isEmpty()) list.add(RouteOption("PHONE_ONLY", "PHONE ONLY (fallback)", 10))

        return list
    }

    private fun determineSystemMode(): SystemMode {
        val roles = nodeRegistry.values.map { it.role.uppercase() }.toSet()
        val hasZero   = roles.contains("PI_ZERO")
        val hasPi5    = roles.contains("PI5")
        val hasLaptop = roles.contains("LAPTOP")

        return when {
            hasZero && hasPi5 && hasLaptop -> SystemMode.AI_CORE
            hasZero && hasPi5              -> SystemMode.EDGE_COMPUTE
            hasZero && hasLaptop           -> SystemMode.PHONE_PIZERO_LAPTOP
            hasPi5  && hasLaptop           -> SystemMode.PHONE_PI5_LAPTOP
            hasZero                        -> SystemMode.MINIMAL_FIELD
            hasPi5                         -> SystemMode.PHONE_PI5
            hasLaptop                      -> SystemMode.EMERGENCY_AI
            else                           -> SystemMode.PHONE_ONLY
        }
    }
    private fun computeOptimalRoute(mode: SystemMode): ActiveRoute {
        val zero = nodeRegistry.values.find { it.role.uppercase() == "PI_ZERO" }?.id
        val pi5  = nodeRegistry.values.find { it.role.uppercase() == "PI5" }?.id
        val lap  = nodeRegistry.values.find { it.role.uppercase() == "LAPTOP" }?.id

        return when (mode) {
            SystemMode.PHONE_ONLY ->
                ActiveRoute("PHONE", "PHONE", "PHONE")

            SystemMode.MINIMAL_FIELD ->
                // Pi Zero streams, phone runs mini model
                ActiveRoute(zero ?: "PHONE", "PHONE", "PHONE")

            SystemMode.PHONE_PI5 ->
                // Pi 5 streams and computes, phone receives
                ActiveRoute(pi5 ?: "PHONE", pi5 ?: "PHONE", "PHONE")

            SystemMode.EMERGENCY_AI ->
                // Phone camera, laptop does heavy inference
                ActiveRoute("PHONE", lap ?: "PHONE", lap ?: "PHONE")

            SystemMode.EDGE_COMPUTE ->
                // Pi Zero streams, Pi 5 computes
                ActiveRoute(zero ?: "PHONE", pi5 ?: "PHONE", "PHONE")

            SystemMode.PHONE_PIZERO_LAPTOP ->
                // Pi Zero streams, phone relays, laptop does AI
                ActiveRoute(zero ?: "PHONE", "PHONE", lap ?: "PHONE")

            SystemMode.PHONE_PI5_LAPTOP ->
                // Pi 5 streams, laptop does heavy AI
                ActiveRoute(pi5 ?: "PHONE", pi5 ?: "PHONE", lap ?: "PHONE")

            SystemMode.AI_CORE ->
                // Full stack: Pi Zero streams, Pi 5 preprocesses, laptop does AI
                ActiveRoute(zero ?: "PHONE", pi5 ?: "PHONE", lap ?: "PHONE")
        }
    }
    @Volatile private var lastRoute: ActiveRoute? = null
    @Volatile var manualOverride: String? = null

    private fun enforceRoute(route: ActiveRoute) {
        if (route == lastRoute) return
        lastRoute = route

        Log.d("ORCH", "Applying route → $route")

        sendCommandSmartAll("STANDBY")

        fun activate(target: String, cmd: String) {
            if (target == "PHONE") return
            sendCommandSmart(target, cmd)
        }

        activate(route.sensor, "START_CAMERA")
        activate(route.compute, "START_COMPUTE")
        activate(route.ai, "START_AI")
    }
    private fun startOrchestratorBrain() {
        CoroutineScope(Dispatchers.IO).launch {
            while (true) {
                try {
                    val mode = determineSystemMode()
                    val route = computeOptimalRoute(mode)
                    enforceRoute(route)
                } catch (e: Exception) {
                    Log.e("ORCH", "brain error: ${e.message}")
                }
                delay(3000)
            }
        }
    }

    fun parseNodeJson(json: String): Node {
        return try {
            val o = JSONObject(json)
            Node(
                id = o.optString("id", "unknown"),
                role = o.optString("role", "UNKNOWN"),
                ip = o.optString("ip", "0.0.0.0"),
                load = o.optDouble("load", 999.0),
                internet = o.optBoolean("internet", false),
                lastSeen = System.currentTimeMillis()
            )
        } catch (e: Exception) {
            Log.e("NODE", "parseNodeJson failed: ${e.message}")
            Node("unknown", "UNKNOWN", "0.0.0.0", 999.0, false, System.currentTimeMillis())
        }
    }

    fun getBestNode(): Node? {
        val now = System.currentTimeMillis()
        return nodeRegistry.values.filter { now - it.lastSeen < 15_000 }
            .maxByOrNull { n: NodeInfo ->
                var score = 0
                if (n.internet) score += 30
                if (n.role.uppercase() == "LAPTOP") score += 50
                score += (10 - n.load.coerceIn(0.0, 10.0)).toInt()
                score
            }?.let { info -> Node(info.id, info.role, info.ip, info.load, info.internet, info.lastSeen) }
    }

    fun computeNodeScore(node: NodeInfo): Int {
        var score = 0
        score += if (node.internet) 30 else 0
        score += if (node.role.uppercase() == "LAPTOP") 50 else 0
        score += (10 - node.load.coerceIn(0.0, 10.0)).toInt()
        return score
    }

    // ---------------- COMMAND DISPATCHER ----------------
    fun sendCommandToNode(node: Node, command: String, params: Map<String, String> = emptyMap()) {
        CoroutineScope(Dispatchers.IO).launch {
            try {
                val payload = mapOf(
                    "command" to command,
                    "params" to params
                )
                val jsonPayload = JSONObject(payload).toString()
                val packetData = jsonPayload.toByteArray()
                val sock = DatagramSocket()
                sock.send(
                    DatagramPacket(packetData, packetData.size, InetAddress.getByName(node.ip), 5006)
                )
                sock.close()
            } catch (e: Exception) {
                e.printStackTrace()
            }
        }
    }

    val client = OkHttpClient.Builder()

    fun replayPortalSubmission(
        node: Node,
        actionUrl: String,
        method: String,
        formData: Map<String, String>
    ) {
        CoroutineScope(Dispatchers.IO).launch {
            try {
                val request = if (method.uppercase() == "POST") {

                    val bodyBuilder = FormBody.Builder()
                    formData.forEach { (k, v) -> bodyBuilder.add(k, v) }

                    Request.Builder()
                        .url(actionUrl)
                        .post(bodyBuilder.build())
                        .header("User-Agent", "Mozilla/5.0")
                        .header("Referer", actionUrl)
                        .build()

                } else {
                    val urlBuilder = StringBuilder(actionUrl)
                    if (formData.isNotEmpty()) {
                        urlBuilder.append("?")
                        urlBuilder.append(formData.entries.joinToString("&") { (k, v) ->
                            "${java.net.URLEncoder.encode(k, "UTF-8")}=${java.net.URLEncoder.encode(v, "UTF-8")}"
                        })
                    }
                    Request.Builder()
                        .url(urlBuilder.toString())
                        .get()
                        .header("User-Agent", "Mozilla/5.0")
                        .header("Referer", actionUrl)
                        .build()
                }

                val response = client.newCall(request).execute()

                try {
                    if (response.isSuccessful) {
                        Log.d("PORTAL", "Submission success → ${node.id}")
                        sendCommandToNode(node, "portal_success")
                    } else {
                        Log.w("PORTAL", "Submission failed ${response.code}")
                    }
                } finally {
                    response.close()
                }

            } catch (e: Exception) {
                Log.e("PORTAL", "Submission crash: ${e.message}", e)
            }
        }
    }

    // ---------------- PERIODIC NODE POLL ----------------
    fun startNodePoller() {
        CoroutineScope(Dispatchers.IO).launch {
            while (true) {
                try {
                    val now = System.currentTimeMillis()
                    // Prune nodes silent for > 6s and tear down their WS
                    val stale = nodeRegistry.entries.filter { now - it.value.lastSeen > 6000 }
                    stale.forEach { (id, _) ->
                        nodeRegistry.remove(id)
                        stopWsConnection(id)
                        Log.w("POLLER", "Pruned stale node: $id")
                    }
                    // Push live counts to Compose state
                    val count = nodeRegistry.size
                    val mode = determineSystemMode()
                    withContext(Dispatchers.Main) {
                        composeNodeCount.intValue = count
                        composeMode.value = mode.name
                        composeStatus.value = "NODES: $count  |  MODE: ${mode.name}"
                    }
                    // Log best node
                    getBestNode()?.let {
                        Log.d("POLLER", "Best: ${it.id} (${it.role}) load=${it.load}")
                    }
                } catch (e: Exception) {
                    Log.e("POLLER", "error: ${e.message}")
                }
                delay(3000)
            }
        }
    }

    fun startBleBootstrap() {

            Log.w("BLE", "Bluetooth not available or disabled"); return

    private fun loadKnownBleDevices() {
        val raw = getSharedPreferences(PREFS, Context.MODE_PRIVATE).getString(PREF_BLE_KNOWN, "") ?: ""
        raw.split(";").filter { it.contains("=") }.forEach {
            val (addr, role) = it.split("=", limit = 2)
            knownBleDevices[addr.trim()] = role.trim()
        }
        Log.d("BLE", "Loaded ${knownBleDevices.size} known BLE devices")
    }

    private fun saveKnownBleDevices() {
        val s = knownBleDevices.entries.joinToString(";") { "${it.key}=${it.value}" }
        getSharedPreferences(PREFS, Context.MODE_PRIVATE).edit().putString(PREF_BLE_KNOWN, s).apply()
    }

    private fun saveKnownBleDevice(address: String, role: String) {
        knownBleDevices[address] = role
        saveKnownBleDevices()
    }

    @SuppressLint("MissingPermission")
    fun startBleScan() {
        if (bleScanActive) return
        bleScanner = bleAdapter?.bluetoothLeScanner
        if (bleScanner == null) { Log.w("BLE", "LE scanner unavailable"); return }
        bleScanActive = true
        bleScanner!!.startScan(bleScanCallback)
        Log.d("BLE", "BLE scan started")
    }

    @SuppressLint("MissingPermission")
    fun stopBleScan() {
        if (!bleScanActive) return
        bleScanActive = false
        try { bleScanner?.stopScan(bleScanCallback) } catch (e: Exception) { Log.w("BLE", "stopScan: ${e.message}") }
    }

    private val bleScanCallback = object : ScanCallback() {
        @SuppressLint("MissingPermission")
        override fun onScanResult(callbackType: Int, result: android.bluetooth.le.ScanResult) {
            val device = result.device ?: return
            val name   = try { device.name } catch (e: Exception) { null } ?: return
            val role   = when {
                name.startsWith(BLE_DEVICE_PIZERO) -> "PIZERO"
                name.startsWith(BLE_DEVICE_PI5)    -> "PI5"
                name.startsWith(BLE_DEVICE_LAPTOP) -> "LAPTOP"
                else -> return
            }
            val addr = device.address
            if (bleConnected[addr] == true) return
            if (knownBleDevices.containsKey(addr)) {
                Log.d("BLE", "Auto-connecting known device $name")
                connectBleDevice(device, role)
                showBleConnectedNotification(name)
            } else {
                runOnUiThread { showBlePairPrompt(device, name, role) }
            }
        }
    }

    @SuppressLint("MissingPermission")
    fun connectBleDevice(device: BluetoothDevice, role: String) {
        if (bleConnected[device.address] == true) return
        val gatt = device.connectGatt(this, false, buildGattCallback(device, role))
        bleGatts[device.address] = gatt
    }

    @SuppressLint("MissingPermission")
    private fun buildGattCallback(device: BluetoothDevice, role: String): BluetoothGattCallback {
        return object : BluetoothGattCallback() {
            override fun onConnectionStateChange(gatt: BluetoothGatt, status: Int, newState: Int) {
                when (newState) {
                    BluetoothProfile.STATE_CONNECTED -> {
                        bleConnected[device.address] = true
                        saveKnownBleDevice(device.address, role)
                        bleChunkBuffers[device.address] = StringBuilder()
                        Log.d("BLE", "Connected ${device.address}, discovering services")
                        gatt.discoverServices()
                    }
                    BluetoothProfile.STATE_DISCONNECTED -> {
                        bleConnected[device.address] = false
                        bleGatts.remove(device.address)
                        bleChunkBuffers.remove(device.address)
                        Log.d("BLE", "Disconnected ${device.address} — scheduling reconnect")
                        // Auto-reconnect after 3s if it was a known device
                        if (knownBleDevices.containsKey(device.address)) {
                            handler.postDelayed({
                                if (bleConnected[device.address] != true) {
                                    Log.d("BLE", "Reconnecting ${device.address}")
                                    val gatt2 = device.connectGatt(
                                        this@MainActivity, false,
                                        buildGattCallback(device, role)
                                    )
                                    bleGatts[device.address] = gatt2
                                }
                            }, 3000L)
                        } else {
                            // Unknown device lost — restart scan to find it again
                            handler.postDelayed({ startBleScan() }, 2000L)
                        }
                    }
                }
            }

            override fun onServicesDiscovered(gatt: BluetoothGatt, status: Int) {
                if (status != BluetoothGatt.GATT_SUCCESS) {
                    Log.w("BLE", "Service discovery failed status=$status — retrying")
                    handler.postDelayed({ gatt.discoverServices() }, 1000L)
                    return
                }
                val svc  = gatt.getService(BLE_SERVICE_UUID) ?: run {
                    Log.w("BLE", "TrackIt service not found on ${device.address}")
                    return
                }
                val stat = svc.getCharacteristic(BLE_STATUS_CHAR_UUID) ?: return
                gatt.setCharacteristicNotification(stat, true)
                val desc = stat.getDescriptor(BLE_NOTIFY_DESC_UUID)
                if (desc != null) {
                    if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.TIRAMISU) {
                        gatt.writeDescriptor(desc, BluetoothGattDescriptor.ENABLE_NOTIFICATION_VALUE)
                    } else {
                        @Suppress("DEPRECATION")
                        desc.value = BluetoothGattDescriptor.ENABLE_NOTIFICATION_VALUE
                        @Suppress("DEPRECATION")
                        gatt.writeDescriptor(desc)
                    }
                }
                // Request larger MTU for chunked payloads
                gatt.requestMtu(512)
                sendCurrentWifiCredentials(gatt)
            }

            override fun onMtuChanged(gatt: BluetoothGatt, mtu: Int, status: Int) {
                Log.d("BLE", "MTU changed to $mtu for ${device.address}")
            }

            // API 33+
            override fun onCharacteristicChanged(
                gatt: BluetoothGatt,
                characteristic: BluetoothGattCharacteristic,
                value: ByteArray
            ) {
                handleBleChunk(gatt, value)
            }

            @Suppress("DEPRECATION")
            @Deprecated("Needed for API < 33")
            override fun onCharacteristicChanged(
                gatt: BluetoothGatt,
                characteristic: BluetoothGattCharacteristic
            ) {
                if (Build.VERSION.SDK_INT < Build.VERSION_CODES.TIRAMISU)
                    handleBleChunk(gatt, characteristic.value ?: return)
            }
        }
    }

    // Reassemble chunked BLE messages then dispatch
    private fun handleBleChunk(gatt: BluetoothGatt, bytes: ByteArray) {
        val addr = gatt.device.address
        val chunk = try { String(bytes, Charsets.UTF_8) } catch (e: Exception) { return }
        val buf = bleChunkBuffers.getOrPut(addr) { StringBuilder() }
        buf.append(chunk)

        // Pi sends complete JSON objects — detect by balanced braces
        val raw = buf.toString().trim()
        if (raw.startsWith("{") && raw.endsWith("}")) {
            // Looks complete — try to parse
            try {
                JSONObject(raw) // validates JSON
                buf.clear()
                handleBleIncoming(gatt, raw.toByteArray(Charsets.UTF_8))
            } catch (e: Exception) {
                // Incomplete — keep buffering
            }
        } else if (buf.length > 8192) {
            // Safety: drop oversized junk
            buf.clear()
            Log.w("BLE", "Dropped oversized BLE buffer for $addr")
        }
    }

    private fun handleBleIncoming(gatt: BluetoothGatt, bytes: ByteArray) {
        val msg = try { String(bytes, Charsets.UTF_8).trim() } catch (e: Exception) { return }
        Log.d("BLE", "Incoming: $msg")
        try {
            val json = JSONObject(msg)
            when (json.optString("type")) {
                "PORTAL_URL"     -> runOnUiThread { showPortalWebView(json.getString("url"), gatt) }
                "WIFI_CONNECTED" -> showToast("Pi connected to WiFi ✓")
                "WIFI_FAILED"    -> showToast("Pi WiFi failed — check password")
                "HEARTBEAT"      -> {
                    val id   = json.optString("id",   "ble_node")
                    val role = json.optString("role", "UNKNOWN")
                    val ip   = json.optString("ip",   "0.0.0.0")
                    val node = NodeInfo(id, role, ip,
                        lastSeen = System.currentTimeMillis(),
                        internet = json.optBoolean("internet", false),
                        load     = json.optDouble("load", 99.0))
                    nodeRegistry[id] = node
                    if (!wsManagers.containsKey(id)) startWsConnection(id, ip, role)
                }
            }
        } catch (e: Exception) { Log.w("BLE", "handleBleIncoming: ${e.message}") }
    }

    @SuppressLint("MissingPermission")
    fun sendBleCredentials(gatt: BluetoothGatt, payload: String) {
        thread(name = "BLE-Send") {
            try {
                val svc  = gatt.getService(BLE_SERVICE_UUID) ?: return@thread
                val char = svc.getCharacteristic(BLE_CMD_CHAR_UUID) ?: return@thread
                payload.toByteArray(Charsets.UTF_8).toList().chunked(180).forEach { chunk ->
                    val arr = chunk.toByteArray()
                    if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.TIRAMISU) {
                        gatt.writeCharacteristic(char, arr, BluetoothGattCharacteristic.WRITE_TYPE_DEFAULT)
                    } else {
                        @Suppress("DEPRECATION")
                        char.value = arr
                        @Suppress("DEPRECATION")
                        gatt.writeCharacteristic(char)
                    }
                    Thread.sleep(30)
                }
                Log.d("BLE", "Credentials sent (${payload.length} chars)")
            } catch (e: Exception) { Log.e("BLE", "sendBleCredentials: ${e.message}") }
        }
    }

// ── WebSocket manager ─────────────────────────────────────────────────────────
// Manages one persistent WebSocket connection to a single Pi node
// Reconnects automatically on disconnect
// Falls back gracefully to UDP if WebSocket unavailable

    inner class NodeWebSocketManager(
        val nodeId: String,
        val nodeIp: String,
        val nodeRole: String
    ) {
        @Volatile var socket: WebSocket? = null
        @Volatile var connected = false
        @Volatile private var running = true
        private var retries = 0
        private val maxRetries = 10

        private val wsClient = OkHttpClient.Builder()
            .connectTimeout(5, TimeUnit.SECONDS)
            .readTimeout(30, TimeUnit.SECONDS)
            .writeTimeout(10, TimeUnit.SECONDS)
            .pingInterval(15, TimeUnit.SECONDS)
            .build()

        fun start() {
            thread(name = "WS-$nodeId") {
                while (running) {
                    try {
                        connect()
                    } catch (e: Exception) {
                        Log.e("WS-$nodeId", "connect error: ${e.message}")
                    }
                    if (running) {
                        val delay = minOf(1000L * (retries + 1), 15000L)
                        Log.d("WS-$nodeId", "Retry in ${delay}ms")
                        Thread.sleep(delay)
                    }
                }
            }
        }

        fun stop() {
            running = false
            socket?.close(1000, "stopped")
            socket = null
            connected = false
            wsConnected[nodeId] = false
            wsConnections.remove(nodeId)
        }

        private fun connect() {
            // Try port 443 first, fallback to 8080
            val ports = listOf(WS_PORT, WS_PORT_FALLBACK)
            for (port in ports) {
                if (!running) return
                try {
                    val url = "ws://$nodeIp:$port/trackit"
                    val request = okhttp3.Request.Builder()
                        .url(url)
                        .header("X-TrackIt-Secret", SECRET)
                        .header("X-TrackIt-NodeId", nodeId)
                        .build()

                    Log.d("WS-$nodeId", "Connecting to $url")

                    socket = wsClient.newWebSocket(request, object : WebSocketListener() {

                        override fun onOpen(webSocket: WebSocket, response: Response) {
                            socket = webSocket
                            connected = true
                            retries = 0
                            wsConnected[nodeId] = true
                            wsConnections[nodeId] = webSocket
                            wsAvailable = true
                            Log.d("WS-$nodeId", "Connected on port $port")
                            webSocket.send("${WS_TYPE_CMD}HELLO|${SECRET}")
                        }

                        override fun onMessage(webSocket: WebSocket, text: String) {
                            handleWsMessage(nodeId, text.toByteArray())
                        }

                        override fun onMessage(webSocket: WebSocket, bytes: ByteString) {
                            handleWsMessage(nodeId, bytes.toByteArray())
                        }

                        override fun onClosing(webSocket: WebSocket, code: Int, reason: String) {
                            webSocket.close(1000, null)
                        }

                        override fun onClosed(webSocket: WebSocket, code: Int, reason: String) {
                            connected = false
                            wsConnected[nodeId] = false
                            wsConnections.remove(nodeId)
                            Log.d("WS-$nodeId", "Closed: $reason")
                            retries++
                        }

                        override fun onFailure(webSocket: WebSocket, t: Throwable, response: Response?) {
                            connected = false
                            wsConnected[nodeId] = false
                            wsConnections.remove(nodeId)
                            Log.w("WS-$nodeId", "Failure on port $port: ${t.message}")
                            retries++
                        }
                    })

                    // Wait to see if connection succeeded
                    Thread.sleep(6000)

                    if (connected) return // success, stay on this port

                } catch (e: Exception) {
                    Log.w("WS-$nodeId", "Port $port failed: ${e.message}")
                }
            }
        }

        fun send(message: String): Boolean {
            return try {
                if (connected && socket != null) {
                    socket!!.send(message)
                    true
                } else false
            } catch (e: Exception) {
                Log.e("WS-$nodeId", "send error: ${e.message}")
                false
            }
        }

        fun sendBytes(bytes: ByteString): Boolean {
            return try {
                if (connected && socket != null) {
                    socket!!.send(bytes)
                    true
                } else false
            } catch (e: Exception) {
                Log.e("WS-$nodeId", "sendBytes error: ${e.message}")
                false
            }
        }
    }

    // ── Active WebSocket managers keyed by nodeId ──────────────────────────────
    private val wsManagers = Collections.synchronizedMap(
        HashMap<String, NodeWebSocketManager>()
    )

    // Start a WebSocket connection to a node
// Called when node appears in heartbeat registry
    fun startWsConnection(nodeId: String, nodeIp: String, nodeRole: String) {
        if (wsManagers.containsKey(nodeId)) return // already managing
        Log.d("WS", "Starting WS manager for $nodeId at $nodeIp")
        val mgr = NodeWebSocketManager(nodeId, nodeIp, nodeRole)
        wsManagers[nodeId] = mgr
        mgr.start()
    }

    // Stop and remove a WebSocket connection
    fun stopWsConnection(nodeId: String) {
        wsManagers[nodeId]?.stop()
        wsManagers.remove(nodeId)
    }

    // Send command to a node — WebSocket first, UDP fallback
    fun sendCommandSmart(nodeId: String, cmd: String) {
        val mgr = wsManagers[nodeId]
        if (mgr != null && mgr.connected) {
            val sent = mgr.send("${WS_TYPE_CMD}$cmd")
            if (sent) {
                Log.d("WS", "Command sent via WebSocket to $nodeId: $cmd")
                return
            }
        }
        // Fallback to UDP
        val node = nodeRegistry[nodeId] ?: return
        worker?.sendUdpFallback(node.ip, "$SECRET|$cmd")
        Log.d("WS", "Command sent via UDP fallback to $nodeId: $cmd")
    }

    // Send to all nodes (WebSocket first, UDP fallback per node)
    fun sendCommandSmartAll(cmd: String) {
        nodeRegistry.keys.forEach { nodeId ->
            sendCommandSmart(nodeId, cmd)
        }
    }

    // ── Central WebSocket message handler ────────────────────────────────────────
// All incoming WS messages from all nodes route through here
    fun handleWsMessage(nodeId: String, bytes: ByteArray) {
        try {
            val raw = String(bytes, Charsets.UTF_8)

            when {
                // ── Heartbeat ─────────────────────────────────────────────────
                raw.startsWith(WS_TYPE_HEARTBEAT) -> {
                    val json = JSONObject(raw.removePrefix(WS_TYPE_HEARTBEAT))
                    val node = NodeInfo(
                        id = json.getString("id"),
                        role = json.optString("role", "UNKNOWN"),
                        ip = json.optString("ip", "0.0.0.0"),
                        lastSeen = System.currentTimeMillis(),
                        internet = json.optBoolean("internet", false),
                        load = json.optDouble("load", 99.0)
                    )
                    nodeRegistry[node.id] = node
                    if (!wsManagers.containsKey(node.id)) {
                        startWsConnection(node.id, node.ip, node.role)
                    }
                }

                // ── Video frame ───────────────────────────────────────────────
                raw.startsWith(WS_TYPE_VIDEO) -> {
                    val payload = bytes.drop(WS_TYPE_VIDEO.length).toByteArray()
                    val bmp = BitmapFactory.decodeByteArray(payload, 0, payload.size)
                    bmp?.let {
                        synchronized(this) {
                            worker?.latestVideoFrame?.recycle()
                            worker?.latestVideoFrame = it
                        }
                    }
                }

                // ── Image chunk ───────────────────────────────────────────────
                raw.startsWith(WS_TYPE_IMAGE) -> {
                    // Forward to worker's image assembler
                    val payload = bytes.drop(WS_TYPE_IMAGE.length).toByteArray()
                    worker?.handleIncomingImageChunk(payload)
                }

                // ── BBox metadata ─────────────────────────────────────────────
                raw.startsWith(WS_TYPE_META) -> {
                    val json = JSONObject(raw.removePrefix(WS_TYPE_META))
                    worker?.handleIncomingMeta(json)
                }

                // ── Log message ───────────────────────────────────────────────
                raw.startsWith(WS_TYPE_LOG) -> {
                    Log.d("WS-NODE-LOG", raw.removePrefix(WS_TYPE_LOG))
                }

                // ── Portal screenshot ─────────────────────────────────────────
                raw.startsWith(WS_TYPE_PORTAL) -> {
                    val json = JSONObject(raw.removePrefix(WS_TYPE_PORTAL))
                    val type = json.optString("subtype")
                    when (type) {
                        "SCREENSHOT" -> {
                            // Base64 screenshot from Pi for remote control UI
                            val b64 = json.getString("data")
                            val imgBytes = android.util.Base64.decode(b64, android.util.Base64.DEFAULT)
                            val bmp = BitmapFactory.decodeByteArray(imgBytes, 0, imgBytes.size)
                            val fields = json.optJSONArray("fields")
                            runOnUiThread {
                                worker?.onPortalScreenshot?.invoke(bmp, fields?.toString())
                            }
                        }
                        "AUTH_SUCCESS" -> {
                            showToast("Pi $nodeId authenticated ✓")
                            Log.d("PORTAL", "Node $nodeId portal auth success")
                        }
                        "AUTH_FAILED" -> {
                            showToast("Pi $nodeId auth failed — check credentials")
                        }
                        "NEEDS_INPUT" -> {
                            val field = json.optString("field", "unknown field")
                            showToast("Pi $nodeId needs input: $field")
                            runOnUiThread {
                                worker?.onPortalNeedsInput?.invoke(nodeId, field)
                            }
                        }
                    }
                }

                // ── ACK ───────────────────────────────────────────────────────
                raw.startsWith(WS_TYPE_ACK) -> {
                    Log.d("WS", "ACK from $nodeId: ${raw.removePrefix(WS_TYPE_ACK)}")
                }
            }
        } catch (e: Exception) {
            Log.e("WS-MSG", "handleWsMessage error: ${e.message}")
        }
    }

        // ----------------- GPS Logging -----------------
        @SuppressLint("MissingPermission")
        private fun startGpsLogging() {
            thread {
                try {
                    locationListener = object : LocationListener {
                        override fun onLocationChanged(location: Location) {
                            try {
                                gpsLog.add(
                                    GPSEntry(
                                        System.currentTimeMillis(),
                                        location.latitude,
                                        location.longitude
                                    )
                                )
                                if (gpsLog.size > 1000) gpsLog.removeAt(0)
                            } catch (e: Exception) {
                                Log.e("FW-GPS", "listener error: ${e.message}")
                            }
                        }

                        override fun onStatusChanged(
                            provider: String?,
                            status: Int,
                            extras: Bundle?
                        ) {
                        }

                        override fun onProviderEnabled(provider: String) {}
                        override fun onProviderDisabled(provider: String) {}
                    }

                    if (ActivityCompat.checkSelfPermission(this@MainActivity,
                            Manifest.permission.ACCESS_FINE_LOCATION) == android.content.pm.PackageManager.PERMISSION_GRANTED) {
                        try {
                            locationManager.requestLocationUpdates(
                                LocationManager.GPS_PROVIDER,
                                1000L,
                                0f,
                                locationListener!!,
                                Looper.getMainLooper()
                            )
                        } catch (e: Exception) {
                            Log.w("FW-GPS", "requestLocationUpdates failed: ${e.message}")
                        }
                    } else {
                        Log.w("FW-GPS", "Location permission not granted")
                    }
                } catch (e: Exception) {
                    Log.e("FW-GPS", "start error: ${e.message}")
                }

                while (running) {
                    try {
                        val loc: Location? = try {
                            locationManager.getLastKnownLocation(LocationManager.GPS_PROVIDER)
                                ?: locationManager.getLastKnownLocation(LocationManager.NETWORK_PROVIDER)
                        } catch (se: SecurityException) {
                            null
                        }

                        loc?.let {
                            gpsLog.add(
                                GPSEntry(
                                    System.currentTimeMillis(),
                                    it.latitude,
                                    it.longitude
                                )
                            )
                            if (gpsLog.size > 1000) gpsLog.removeAt(0)
                        }
                    } catch (e: Exception) {
                        Log.e("FW-GPS", "poll error: ${e.message}")
                    }
                    Thread.sleep(1500)
                }
            }
        }

        fun getNearestGps(ts: Long): GPSEntry? {
            synchronized(gpsLog) {
                if (gpsLog.isEmpty()) return null
                return gpsLog.minByOrNull { kotlin.math.abs(it.ts - ts) }
            }
        }

        // ----------------- Swarm heartbeat listener -----------------
        private fun startSwarmHeartbeatListener() {
            thread {
                try {
                    DatagramSocket(NODE_HEARTBEAT_PORT).use { socket ->
                        val buf = ByteArray(4096)
                        while (running) {
                            try {
                                val packet = DatagramPacket(buf, buf.size)
                                socket.receive(packet)
                                val msg = String(packet.data, 0, packet.length)
                                val json = JSONObject(msg)
                                val node = NodeInfo(
                                    id = json.getString("id"),
                                    role = json.optString("role", "UNKNOWN"),
                                    ip = json.optString("ip", packet.address.hostAddress ?: "0.0.0.0"),
                                    lastSeen = System.currentTimeMillis(),
                                    internet = json.optBoolean("internet", false),
                                    load = json.optDouble("load", 99.0)
                                )
                                nodeRegistry[node.id] = node

// Auto-start WebSocket to this node if not already connected
                                if (!wsManagers.containsKey(node.id)) {
                                    this@MainActivity.startWsConnection(node.id, node.ip, node.role)
                                }

// cleanup stale nodes
                                val now = System.currentTimeMillis()
                                val toRemove = nodeRegistry.keys.filter { nid ->
                                    now - (nodeRegistry[nid]?.lastSeen ?: 0L) > 6000L
                                }
                                toRemove.forEach { nodeRegistry.remove(it) }
                            } catch (e: Exception) {
                                Log.e("FW-SWARM", "heartbeat parse error: ${e.message}")
                                Thread.sleep(50)
                            }
                        }
                    }
                } catch (e: Exception) {
                    Log.e("FW-SWARM", "socket error: ${e.message}")
                }
            }
        }

        // ----------------- Discovery (broadcast) -----------------
        private fun startDiscovery() {
            thread {
                try {
                    DatagramSocket(DISCOVERY_PORT).use { socket ->
                        socket.broadcast = true
                        // reply listener
                        thread {
                            while (running) {
                                try {
                                    val buf = ByteArray(1024) // allocate per iteration
                                    val packet = DatagramPacket(buf, buf.size)
                                    socket.receive(packet)
                                    val msg = String(packet.data, 0, packet.length)
                                    if (msg.startsWith("OK|PI|")) {
                                        piIp = msg.substring(6)
                                        Log.d("FW-DISC", "Found PI: $piIp")
                                    }
                                } catch (e: Exception) {
                                    // ignore
                                }
                            }
                        }

                        // broadcast loop
                        while (running) {
                            try {
                                val broadcastAddr = InetAddress.getByName("255.255.255.255")
                                val discoveryMsg = "$SECRET|DISCOVER".toByteArray(Charsets.UTF_8)
                                val packet = DatagramPacket(
                                    discoveryMsg,
                                    discoveryMsg.size,
                                    broadcastAddr,
                                    DISCOVERY_PORT
                                )
                                socket.send(packet)
                            } catch (e: Exception) {
                                Log.e("FW-DISC", "broadcast error: ${e.message}")
                            }
                            Thread.sleep(3000)
                        }
                    }
                } catch (e: Exception) {
                    Log.e("FW-DISC", "discovery socket error: ${e.message}")
                }
            }
        }

        // ----------------- Metadata listener (bbox messages) -----------------
        private fun startMetadataListener() {
            thread(name = "UDP-Meta-Listener") {
                DatagramSocket(null).use { socket ->
                    try {
                        socket.reuseAddress = true
                        socket.bind(InetSocketAddress(META_PORT))
                        socket.soTimeout = 500 // allow periodic running check
                        val buf = ByteArray(8192)

                        while (running) {
                            try {
                                val packet = DatagramPacket(buf, buf.size)
                                socket.receive(packet)
                                val msg = String(packet.data, 0, packet.length, Charsets.UTF_8)

                                val json = try {
                                    JSONObject(msg)
                                } catch (e: Exception) {
                                    Log.w("FW-META", "Invalid JSON: ${e.message}")
                                    continue
                                }

                                if (json.optString("type") != "bbox") continue

                                val fname = json.getString("filename")
                                val bboxArray = json.getJSONArray("bbox")
                                val box = BBox(
                                    bboxArray.getInt(0),
                                    bboxArray.getInt(1),
                                    bboxArray.getInt(2),
                                    bboxArray.getInt(3),
                                    json.getString("classname"),
                                    json.getInt("trackId")
                                )

                                // Thread-safe addition to detection
                                synchronized(detections) {
                                    val found = detections.find { it.filename == fname }
                                    found?.boxes?.add(box)
                                }

                            } catch (e: SocketTimeoutException) {
                                // ignore timeout, allows loop to check running
                            } catch (e: Exception) {
                                Log.e("FW-META", "BBox receive error: ${e.message}")
                                Thread.sleep(10)
                            }
                        }
                    } catch (e: Exception) {
                        Log.e("FW-META", "Meta socket failed: ${e.message}")
                    }
                }
            }
        }

        // ----------------- Image chunk receiver -----------------
        private fun startImageListener() {
            thread(name = "UDP-Image-Listener") {
                try {
                    DatagramSocket(null).use { socket ->
                        socket.reuseAddress = true
                        socket.bind(InetSocketAddress(IMAGE_PORT))
                        socket.soTimeout = 500

                        val imageBuffers = HashMap<String, ImageAssembler>()

                        while (running) {
                            try {
                                val buf = ByteArray(65507)
                                val packet = DatagramPacket(buf, buf.size)
                                socket.receive(packet)

                                // find first 6 '|' delimiters
                                val pipeIdxs = ArrayList<Int>(6)
                                for (i in 0 until packet.length) {
                                    if (packet.data[i].toInt() == '|'.code) {
                                        pipeIdxs.add(i)
                                        if (pipeIdxs.size == 6) break
                                    }
                                }
                                if (pipeIdxs.size < 6) continue

                                val headerEnd = pipeIdxs[5] + 1
                                val headerStr = String(packet.data, 0, headerEnd, Charsets.UTF_8)
                                val parts = headerStr.split("|")
                                if (parts.size < 6 || parts[0] != "IMG") continue

                                val fname = parts[1]
                                val seq = parts[2].toIntOrNull() ?: continue
                                val total = parts[3].toIntOrNull() ?: continue
                                val payloadLen = parts[5].toIntOrNull() ?: continue

                                if (headerEnd + payloadLen > packet.length) continue

                                val assembler = imageBuffers.getOrPut(fname) {
                                    ImageAssembler(total)
                                }

                                val payload =
                                    packet.data.copyOfRange(headerEnd, headerEnd + payloadLen)
                                assembler.insert(seq, payload)

                                if (assembler.isComplete()) {
                                    val fullImage = assembler.build()
                                    imageBuffers.remove(fname)

                                    val bmp =
                                        BitmapFactory.decodeByteArray(fullImage, 0, fullImage.size)
                                            ?: continue

                                    val partsF = fname.split("_")
                                    val ts = partsF.getOrNull(0)?.toLongOrNull()
                                        ?: System.currentTimeMillis()
                                    val tid = partsF.getOrNull(1)?.toIntOrNull() ?: 0
                                    val cls = partsF.getOrNull(2) ?: "object"

                                    val gps = getNearestGps(System.currentTimeMillis())
                                    val detection = Detection(
                                        fname,
                                        cls,
                                        tid,
                                        ts,
                                        gps?.lat,
                                        gps?.lon,
                                        fullImage,
                                        bmp
                                    )

                                    synchronized(detections) {
                                        detections.add(0, detection)
                                        if (detections.size > 200) detections.removeAt(detections.lastIndex)
                                    }

                                    saveImageToGallery(fname, bmp)

                                    if (getSupabaseEnabled()) {
                                        enqueueLogForUpload(fname, detection)
                                    }
                                }

                            } catch (e: SocketTimeoutException) {
                                // allows loop exit
                            } catch (e: Exception) {
                                Log.e("FW-IMG", "img recv error: ${e.message}")
                                Thread.sleep(10)
                            }
                        }
                    }
                } catch (e: Exception) {
                    Log.e("FW-IMG", "image socket fail: ${e.message}")
                }
            }
        }

        private fun saveImageToGallery(fname: String, bitmap: Bitmap) {
            try {
                val filename = "$fname.jpg"
                val resolver = this@MainActivity.contentResolver
                val contentValues = ContentValues().apply {
                    put(MediaStore.MediaColumns.DISPLAY_NAME, filename)
                    put(MediaStore.MediaColumns.MIME_TYPE, "image/jpeg")
                    if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.Q) {
                        put(MediaStore.MediaColumns.RELATIVE_PATH, "Pictures/MemoryRetrieve")
                    }
                }
                val uri =
                    resolver.insert(MediaStore.Images.Media.EXTERNAL_CONTENT_URI, contentValues)
                uri?.let {
                    resolver.openOutputStream(it)?.use { out ->
                        bitmap.compress(Bitmap.CompressFormat.JPEG, 90, out)

            }
        }

        private fun startSupabaseFlushLoop() {
            thread {

                            }

                    } catch (e: Exception) {

                    }
                    Thread.sleep(10_000) // every 10s if enabled

        // Very small best-effort HTTP POST to Supabase (Project URL base must be set by user; we use path /rest/v1/logs as example).
        // WARNING: you must configure correct endpoint/headers on Supabase (this is a template).
        private fun getSupabaseBaseUrl(): String? {
            return prefs.getString(PREF_SUPABASE_URL, null)
        }

        private fun getSupabaseApiKey(): String? {
            return prefs.getString(PREF_SUPABASE_KEY, null)

        private fun flushUploadQueueToSupabase() {
            thread(name = "SupabaseUploader") {
                if (!getSupabaseEnabled()) return@thread
                if (uploadQueue.isEmpty()) return@thread

                val baseUrl = getSupabaseBaseUrl()
                val apiKey = getSupabaseApiKey()

                if (baseUrl.isNullOrBlank() || apiKey.isNullOrBlank()) {
                    Log.w("FW-SUPA", "Supabase not configured")
                    return@thread
                }

                val queueCopy = ArrayList(uploadQueue)

                for (entry in queueCopy) {
                    try {
                        val url = URL("$baseUrl/rest/v1/logs")
                        val conn = url.openConnection() as HttpURLConnection

                        conn.requestMethod = "POST"
                        conn.setRequestProperty("Content-Type", "application/json")
                        conn.setRequestProperty("apikey", apiKey)
                        conn.setRequestProperty("Authorization", "Bearer $apiKey")
                        conn.doOutput = true
                        conn.connectTimeout = 5000
                        conn.readTimeout = 5000

                        conn.outputStream.use {
                            it.write(entry.toString().toByteArray())
                        }

                        val code = conn.responseCode
                        if (code in 200..299) {
                            uploadQueue.remove(entry)
                            Log.d("FW-SUPA", "Uploaded: ${entry.optString("filename")}")
                        } else {
                            Log.w("FW-SUPA", "Upload failed ($code), retry later")
                        }

                        conn.disconnect()

                    } catch (e: Exception) {
                        Log.w("FW-SUPA", "Upload error: ${e.message}")
                        break   // stop loop, retry later
                    }
                }
            }
        }

        private fun getSupabaseEnabled(): Boolean {
            return prefs.getBoolean(PREF_SUPABASE_ENABLED, false)
        }

        private fun startUdpListeners() {
            // Video frame listener (port 5001)
            thread(name = "UDP-Video-Listener") {
                try {
                    DatagramSocket(null).use { socket ->
                        socket.reuseAddress = true
                        socket.bind(InetSocketAddress(VIDEO_PORT))
                        socket.soTimeout = 500
                        val buf = ByteArray(65507)
                        while (running) {
                            try {
                                val packet = DatagramPacket(buf, buf.size)
                                socket.receive(packet)
                                val data = packet.data.copyOf(packet.length)
                                // Strip secret prefix if present
                                val payload = if (String(data).startsWith(SECRET)) {
                                    data.drop(SECRET.length + 1).toByteArray()
                                } else data
                                val bmp = BitmapFactory.decodeByteArray(payload, 0, payload.size)
                                bmp?.let {
                                    synchronized(this) {
                                        latestVideoFrame?.recycle()
                                        latestVideoFrame = it
                                    }
                                }
                            } catch (e: SocketTimeoutException) {
                                // normal
                            } catch (e: Exception) {
                                Log.e("UDP-VID", "recv error: ${e.message}")
                                Thread.sleep(10)
                            }
                        }
                    }
                } catch (e: Exception) {
                    Log.e("UDP-VID", "socket fail: ${e.message}")
                }
            }

            // Log listener (port 5002)
            thread(name = "UDP-Log-Listener") {
                try {
                    DatagramSocket(null).use { socket ->
                        socket.reuseAddress = true
                        socket.bind(InetSocketAddress(LOG_PORT))
                        socket.soTimeout = 500
                        val buf = ByteArray(4096)
                        while (running) {
                            try {
                                val packet = DatagramPacket(buf, buf.size)
                                socket.receive(packet)
                                val msg = String(packet.data, 0, packet.length).trim()
                                val clean = if (msg.startsWith(SECRET)) msg.removePrefix("$SECRET|") else msg
                                Log.d("PI-LOG", clean)

                                Thread.sleep(10)
                            }
                        }
                    }

                    Log.e("UDP-LOG", "socket fail: ${e.message}")
                }
            }

        // Expose image chunk handler for WebSocket routing
        fun handleIncomingImageChunk(payload: ByteArray) {
            // Reuses existing assembler logic
            // Wrap payload as if it came from UDP
            // The assembler doesn't care about transport
            try {
                val pipeIdxs = ArrayList<Int>(6)
                for (i in payload.indices) {
                    if (payload[i].toInt() == '|'.code) {
                        pipeIdxs.add(i)
                        if (pipeIdxs.size == 6) break
                    }
                }
                if (pipeIdxs.size < 6) return

                val headerEnd = pipeIdxs[5] + 1
                val headerStr = String(payload, 0, headerEnd, Charsets.UTF_8)
                val parts = headerStr.split("|")
                if (parts.size < 6 || parts[0] != "IMG") return

                val fname = parts[1]
                val seq = parts[2].toIntOrNull() ?: return
                val total = parts[3].toIntOrNull() ?: return
                val payloadLen = parts[5].toIntOrNull() ?: return

                if (headerEnd + payloadLen > payload.size) return

                val assembler = imageAssemblers.getOrPut(fname) { ImageAssembler(total) }
                val chunk = payload.copyOfRange(headerEnd, headerEnd + payloadLen)
                assembler.insert(seq, chunk)

                if (assembler.isComplete()) {
                    val fullImage = assembler.build()
                    imageAssemblers.remove(fname)
                    val bmp = BitmapFactory.decodeByteArray(fullImage, 0, fullImage.size) ?: return
                    val partsF = fname.split("_")
                    val ts = partsF.getOrNull(0)?.toLongOrNull() ?: System.currentTimeMillis()
                    val tid = partsF.getOrNull(1)?.toIntOrNull() ?: 0
                    val cls = partsF.getOrNull(2) ?: "object"
                    val gps = getNearestGps(System.currentTimeMillis())
                    val detection = Detection(fname, cls, tid, ts, gps?.lat, gps?.lon, fullImage, bmp)
                    synchronized(detections) {
                        detections.add(0, detection)
                        if (detections.size > 200) detections.removeAt(detections.lastIndex)
                    }
                    saveImageToGallery(fname, bmp)
                    if (getSupabaseEnabled()) enqueueLogForUpload(fname, detection)
                }
            } catch (e: Exception) {
                Log.e("WS-IMG", "chunk error: ${e.message}")
            }
        }

        // ----------------- Pi AP watcher: scans for Pi Zero AP when offline and prompts phone UI -----------------
        private fun startPiApWatcher() {
            thread {
                try {
                    val wifiMgr =
                        applicationContext.getSystemService(Context.WIFI_SERVICE) as WifiManager
                    while (running) {
                        try {
                            // Only actively scan if no internet or no known wifi
                            if (!hasInternet()) {
                                try {
                                    wifiMgr.startScan()
                                } catch (_: Exception) {
                                }
                                val results = try {
                                    wifiMgr.scanResults
                                } catch (_: Exception) {
                                    emptyList<android.net.wifi.ScanResult>()
                                }
                                val piAp = results.firstOrNull { r: android.net.wifi.ScanResult ->
                                    val s = r.SSID ?: ""
                                    s.startsWith("PI_ZERO") || s.startsWith("PI_AP") || s.startsWith(
                                        "TRACKPI"
                                    ) || s.startsWith("PI5_RECOVERY")
                                }
                                if (piAp != null) {
                                    Log.d("FW-AP", "Found Pi AP: ${piAp.SSID}")
                                    // ask MainActivity to show popup
                                    this@MainActivity.runOnUiThread {
                                        showPiConnectPopup(piAp.SSID ?: "PI")
                                    }
                                    // Wait a bit so we don't spam UI
                                    Thread.sleep(8000)
                                }
                            }
                        } catch (e: Exception) {
                            Log.e("FW-AP", "ap watcher error: ${e.message}")
                        }
                        Thread.sleep(3000)
                    }
                } catch (e: Exception) {
                    Log.e("FW-AP", "ap watcher setup failed: ${e.message}")
                }
            }
        }

        // ----------------- Simple network brain loop to enable offline scanning / recovery
        private var lastInternetState: Boolean? = null

        private fun networkBrainLoop() {
            thread {
                while (running) {
                    try {
                        val internet = hasInternet()
                        if (internet != lastInternetState) {
                            lastInternetState = internet
                            if (!internet) {
                                Log.w("FW-NETB", "Internet LOST — forcing phone fallback evaluation")
                                evaluateRouting()
                            } else {
                                Log.i("FW-NETB", "Internet RESTORED — re-evaluating routing")
                                // Give nodes 3s to re-announce themselves via heartbeat
                                Thread.sleep(3000)
                                evaluateRouting()
                            }
                        }
                    } catch (e: Exception) {
                        Log.e("FW-NETB", "net brain error: ${e.message}")
                    }
                    Thread.sleep(5000)
                }
            }
        }

            thread {

        private fun evaluateRouting() {
            val nodes: List<NodeInfo> = this@MainActivity.nodeRegistry.values.toList()

            // If nothing alive → phone-only mode
            if (nodes.isEmpty()) {
                activatePhoneFallback()
                return
            }

            val compute = selectBestComputeNode(nodes)
            val stream = selectBestStreamNode(nodes)

            if (compute == null || stream == null) {
                activatePhoneFallback()
                return
            }

            val newRoute = WorkerRoute(
                streamNode = stream.id,
                computeNode = compute.id,
                outputNode = "PHONE"
            )

            if (activeRoute != newRoute) {
                applyRoute(newRoute)
            }
        }

        private fun applyRoute(route: WorkerRoute) {
            Log.i("ORCH", "Applying route: $route")
            activeRoute = route
            deactivatePhoneFallback()
            this@MainActivity.sendCommandSmart(route.streamNode, "STREAM_START")
            this@MainActivity.sendCommandSmart(route.computeNode, "COMPUTE_START")
        }

        private fun probeLatency(ip: String): Int {
            return try {
                val start = System.currentTimeMillis()
                val addr = InetAddress.getByName(ip)
                if (addr.isReachable(1000)) (System.currentTimeMillis() - start).toInt() else 999
            } catch (_: Exception) {
                999
            }
        }

        private fun selectBestComputeNode(nodes: List<NodeInfo>): NodeInfo? {
            return nodes
                .filter { it.role.uppercase() in listOf("LAPTOP", "PI5") }
                .maxByOrNull { computeNodeScore(it) }
        }

        private fun selectBestStreamNode(nodes: List<NodeInfo>): NodeInfo? {
            return nodes
                .filter { it.role.uppercase() in listOf("PI0", "PIZERO", "PI_ZERO") }
                .minByOrNull { probeLatency(it.ip) }
        }

        private fun sendCommandToNode(nodeId: String, cmd: String) {
            this@MainActivity.sendCommandSmart(nodeId, cmd)
        }

        @Volatile private var tflite: org.tensorflow.lite.Interpreter? = null
        @Volatile private var tfliteReady = false
        private val tfliteLabels = mutableListOf<String>()
        private val TFLITE_INPUT_SIZE = 300

        private fun loadTfliteModel() {
            thread(name = "TFLite-Loader") {
                try {
                    val afd = this@MainActivity.assets.openFd("detect.tflite")
                    val stream = afd.createInputStream()
                    val modelBytes = stream.readBytes()
                    stream.close()
                    val buf = java.nio.ByteBuffer.allocateDirect(modelBytes.size).apply {
                        order(java.nio.ByteOrder.nativeOrder())
                        put(modelBytes)
                        rewind()
                    }
                    tflite = org.tensorflow.lite.Interpreter(buf, org.tensorflow.lite.Interpreter.Options().apply { setNumThreads(2) })
                    try {
                        this@MainActivity.assets.open("labelmap.txt").bufferedReader()
                            .forEachLine { l -> if (l.isNotBlank()) tfliteLabels.add(l.trim()) }
                    } catch (_: Exception) {
                        tfliteLabels.addAll(listOf("person","car","bicycle","dog","cat","bottle","chair","phone","laptop","bag"))
                    }
                    tfliteReady = true
                    Log.i("TFLite", "Model loaded, ${tfliteLabels.size} labels")
                } catch (e: Exception) {
                    Log.w("TFLite", "Model load failed — inference disabled: ${e.message}")
                }
            }
        }

        private fun activatePhoneFallback() {

        }

        private fun handlePhoneFrame(image: ImageProxy) {
            try {
                val buffer = image.planes[0].buffer
                val bytes = ByteArray(buffer.remaining()).also { buffer.get(it) }
                val rawBmp = BitmapFactory.decodeByteArray(bytes, 0, bytes.size) ?: return

                val displayBmp = rawBmp.copy(Bitmap.Config.ARGB_8888, true)
                synchronized(this) {
                    latestVideoFrame?.recycle()
                    latestVideoFrame = displayBmp
                }

                if (!tfliteReady || tflite == null) { rawBmp.recycle(); return }

                val resized = Bitmap.createScaledBitmap(rawBmp, TFLITE_INPUT_SIZE, TFLITE_INPUT_SIZE, true)
                rawBmp.recycle()

                val inputBuf = java.nio.ByteBuffer.allocateDirect(TFLITE_INPUT_SIZE * TFLITE_INPUT_SIZE * 3).apply {
                    order(java.nio.ByteOrder.nativeOrder())
                }
                val pixels = IntArray(TFLITE_INPUT_SIZE * TFLITE_INPUT_SIZE)
                resized.getPixels(pixels, 0, TFLITE_INPUT_SIZE, 0, 0, TFLITE_INPUT_SIZE, TFLITE_INPUT_SIZE)
                resized.recycle()
                for (px in pixels) {
                    inputBuf.put(((px shr 16) and 0xFF).toByte())
                    inputBuf.put(((px shr 8)  and 0xFF).toByte())
                    inputBuf.put(( px         and 0xFF).toByte())
                }
                inputBuf.rewind()

                val outBoxes   = Array(1) { Array(10) { FloatArray(4) } }
                val outClasses = Array(1) { FloatArray(10) }
                val outScores  = Array(1) { FloatArray(10) }
                val outCount   = FloatArray(1)
                tflite!!.runForMultipleInputsOutputs(
                    arrayOf(inputBuf),
                    mapOf(0 to outBoxes, 1 to outClasses, 2 to outScores, 3 to outCount)
                )

                val count = outCount[0].toInt().coerceIn(0, 10)
                val ts = System.currentTimeMillis()
                val gps = getNearestGps(ts)

                for (i in 0 until count) {
                    val score = outScores[0][i]
                    if (score < 0.45f) continue
                    val label = tfliteLabels.getOrElse(outClasses[0][i].toInt()) { "obj" }
                    val box = outBoxes[0][i]
                    val x1 = (box[1] * TFLITE_INPUT_SIZE).toInt()
                    val y1 = (box[0] * TFLITE_INPUT_SIZE).toInt()
                    val x2 = (box[3] * TFLITE_INPUT_SIZE).toInt()
                    val y2 = (box[2] * TFLITE_INPUT_SIZE).toInt()
                    val jpegOut = java.io.ByteArrayOutputStream()
                    displayBmp.compress(Bitmap.CompressFormat.JPEG, 70, jpegOut)
                    val fname = "${ts}_${i}_${label}"
                    val detection = Detection(
                        filename  = fname,
                        className = label,
                        trackId   = i,
                        ts        = ts,
                        lat       = gps?.lat,
                        lon       = gps?.lon,
                        rawBytes  = jpegOut.toByteArray(),
                        bitmap    = displayBmp.copy(Bitmap.Config.ARGB_8888, false),
                        boxes     = mutableListOf(BBox(x1, y1, x2, y2, label, i))
                    )
                    synchronized(detections) {
                        detections.removeAll { it.className == label && it.trackId == i }
                        detections.add(0, detection)
                        if (detections.size > 200) detections.removeAt(detections.lastIndex)
                    }
                    if (getSupabaseEnabled()) enqueueLogForUpload(fname, detection)
                }
            } catch (e: Exception) {
                Log.e("PHONE-TF", "Frame error: ${e.message}")
            } finally {
                image.close()
            }
        }

        private fun startFailoverMonitor() {

        private fun checkNodeHealth() {
            val now = System.currentTimeMillis()
            val deadNodes = this@MainActivity.nodeRegistry.values.filter {
                now - it.lastSeen > 6000
            }

            deadNodes.forEach {
                Log.w("FAILOVER", "Node lost: ${it.id}")
                nodeRegistry.remove(it.id)
                // Clean up WebSocket for dead node
                this@MainActivity.stopWsConnection(it.id)
            }

            if (deadNodes.isNotEmpty()) {
                Log.w("FAILOVER", "Node(s) lost: ${deadNodes.joinToString { it.id }}")
                // Immediately adjust routing using the existing brain
                evaluateRouting()
            }
        }

        private fun hasInternet(): Boolean {
            try {
                val cm = getSystemService(Context.CONNECTIVITY_SERVICE) as ConnectivityManager
                val net = cm.activeNetwork ?: return false
                val caps = cm.getNetworkCapabilities(net) ?: return false
                return caps.hasCapability(android.net.NetworkCapabilities.NET_CAPABILITY_INTERNET)
            } catch (e: Exception) {
                return false
            }
        }

        private fun computeNodeScore(n: NodeInfo): Int {
            var score = 0

            if (n.internet) score += 25
            if (n.role.uppercase() == "LAPTOP") score += 50
            if (n.role.uppercase() == "PI5") score += 35

            val cpuPenalty = n.load.coerceIn(0.0, 10.0)
            score += (10 - cpuPenalty).toInt()

            val latency = probeLatency(n.ip)
            score -= (latency / 40).toInt()

            return score
        }

    // Programmatic connect using WifiNetworkSpecifier (Android Q+). Binds process to network for portal auth.
    private var boundNetworkCallback: ConnectivityManager.NetworkCallback? = null
    private var wifiChangeReceiver: android.content.BroadcastReceiver? = null
    @SuppressLint("MissingPermission")
    private fun connectToPiApUsingSpecifier(ssid: String) {
        if (Build.VERSION.SDK_INT < Build.VERSION_CODES.Q) {
            startActivity(Intent(android.provider.Settings.ACTION_WIFI_SETTINGS))
            return
        }
        try {
            val spec = WifiNetworkSpecifier.Builder()
                .setSsid(ssid)
                .build()
            val req = NetworkRequest.Builder()
                .addTransportType(android.net.NetworkCapabilities.TRANSPORT_WIFI)
                .setNetworkSpecifier(spec)
                .build()
            val cm = getSystemService(Context.CONNECTIVITY_SERVICE) as ConnectivityManager
            // unregister previous
            try { boundNetworkCallback?.let { cm.unregisterNetworkCallback(it) } } catch (_: Exception) {}
            boundNetworkCallback = object : ConnectivityManager.NetworkCallback() {
                override fun onAvailable(network: Network) {
                    try {
                        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.M) {
                            cm.bindProcessToNetwork(network)
                        } else {
                            @Suppress("DEPRECATION")
                            ConnectivityManager.setProcessDefaultNetwork(network)
                        }
                        showToast("Bound to $ssid for portal auth")
                        // notify Pi that phone bound (simple UDP ping)
                        thread {
                            try {
                                DatagramSocket().use { s ->
                                    val msg = "$SECRET|PHONE_BOUND".toByteArray()
                                    s.send(DatagramPacket(msg, msg.size, InetAddress.getByName("255.255.255.255"), COMMAND_PORT))
                                }
                            } catch (_: Exception) {}
                        }
                    } catch (e: Exception) {
                        Log.w("MA-WIFI", "bind failed: ${e.message}")
                    }
                }
                override fun onUnavailable() {
                    showToast("Failed to connect to $ssid")
                }
                override fun onLost(network: Network) {
                    showToast("$ssid disconnected")
                    try {
                        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.M) {
                            cm.bindProcessToNetwork(null)
                        } else {
                            @Suppress("DEPRECATION")
                            ConnectivityManager.setProcessDefaultNetwork(null)
                        }
                    } catch (_: Exception) {}
                }
            }
            cm.requestNetwork(req, boundNetworkCallback!!)
        } catch (e: Exception) {
            Log.e("MA-WIFI", "connect spec err: ${e.message}")
            startActivity(Intent(android.provider.Settings.ACTION_WIFI_SETTINGS))
        }
    }

        private val COMMAND_PORT = 5005

            super.onCreate(savedInstanceState)
            try {
                portalHtml = intent.getStringExtra("portal_html")
                fromIp = intent.getStringExtra("from_ip")
            } catch (_: Exception) {
            }

            layout.orientation = LinearLayout.VERTICAL
            val web = WebView(this)
            web.layoutParams =
                LinearLayout.LayoutParams(LinearLayout.LayoutParams.MATCH_PARENT, 0, 1f)

                override fun onPageFinished(view: WebView?, url: String?) {
                    super.onPageFinished(view, url)
                }
            }

            val btn = Button(this)
            btn.text = "Done — Notify Pi"
            btn.setOnClickListener {
                thread {
                    try {
                        val target = fromIp ?: return@thread
                        DatagramSocket().use { s ->
                            val msg = "$SECRET|PORTAL_AUTH_DONE".toByteArray(Charsets.UTF_8)
                            s.send(
                                DatagramPacket(
                                    msg,
                                    msg.size,
                                    InetAddress.getByName(target),
                                    COMMAND_PORT
                                )
                            )
                        }
                    } catch (e: Exception) {
                        Log.e("PORTAL-ACT", "notify pi failed: ${e.message}")
                    }
                }
                finish()
            }


            layout.addView(web)
            layout.addView(btn)
            setContentView(layout)

            if (!portalHtml.isNullOrBlank()) {
                try {
                    web.loadDataWithBaseURL(null, portalHtml!!, "text/html", "utf-8", null)
                } catch (e: Exception) {
                    Log.e("PORTAL-ACT", "load html failed: ${e.message}")
                }
            } else {
                web.loadUrl("about:blank")
            }
        }
    }

    // ----------------- Utility HMAC (if needed later) -----------------
    private fun hmacSha256Hex(key: ByteArray, msg: ByteArray): String {
        return try {
            val mac = Mac.getInstance("HmacSHA256")
            val sk = SecretKeySpec(key, "HmacSHA256")
            mac.init(sk)
            val digest = mac.doFinal(msg)
            digest.joinToString("") { "%02x".format(it) }
        } catch (e: Exception) {
            Log.e("MA-HMAC", "err: ${e.message}")
            ""
        }
    }

    @androidx.compose.runtime.Composable
    fun ChatbotScreen() {
        val messages = androidx.compose.runtime.remember {
            androidx.compose.runtime.mutableStateListOf<Pair<String, String>>() // role, text
        }
        var inputText by androidx.compose.runtime.remember {
            androidx.compose.runtime.mutableStateOf("")
        }
        var isThinking by androidx.compose.runtime.remember {
            androidx.compose.runtime.mutableStateOf(false)
        }
        val listState = androidx.compose.foundation.lazy.rememberLazyListState()
        val scope = androidx.compose.runtime.rememberCoroutineScope()

        // Build context string from live system state
        fun buildSystemContext(): String {
            val mode = composeMode.value
            val nodeCount = composeNodeCount.intValue
            val nodeDetails = nodeRegistry.values.joinToString(", ") {
                "${it.id}(${it.role}, load=${"%.1f".format(it.load)}, internet=${it.internet})"
            }
            val detectionCount = synchronized(detections) { detections.size }
            val lastGps = worker?.gpsLog?.lastOrNull()
            val gpsStr = if (lastGps != null) "${"%.5f".format(lastGps.lat)},${"%.5f".format(lastGps.lon)}" else "no fix"
            return """
You are TrackIt AI, the onboard assistant for a real-time object detection and GPS tracking system.
Current system state:
- Mode: $mode
- Active nodes: $nodeCount ($nodeDetails)
- Detections in memory: $detectionCount
- Last GPS: $gpsStr
- Transport: WebSocket (port 443 primary, 8080 fallback), UDP same-subnet fallback
Answer questions about node status, detection results, system modes, and routing. 
Be concise. If asked to send a command, reply with exactly: CMD:<nodeId>:<command>
""".trimIndent()
        }

        fun sendMessage(userText: String) {
            if (userText.isBlank() || isThinking) return
            messages.add("user" to userText)
            inputText = ""
            isThinking = true

            scope.launch(Dispatchers.IO) {
                try {
                    val msgArray = org.json.JSONArray()
                    // Include last 10 messages for context
                    messages.takeLast(10).forEach { (role, text) ->
                        msgArray.put(org.json.JSONObject().apply {
                            put("role", role)
                            put("content", text)
                        })
                    }
                    val body = org.json.JSONObject().apply {
                        put("model", "claude-sonnet-4-20250514")
                        put("max_tokens", 400)
                        put("system", buildSystemContext())
                        put("messages", msgArray)
                    }
                    val req = okhttp3.Request.Builder()
                        .url("https://api.anthropic.com/v1/messages")
                        .post(okhttp3.RequestBody.create(
                            "application/json".toMediaType(),
                            body.toString()
                        ))
                        .header("x-api-key", getSharedPreferences(PREFS, Context.MODE_PRIVATE)
                            .getString("anthropic_key", "") ?: "")
                        .header("anthropic-version", "2023-06-01")
                        .header("content-type", "application/json")
                        .build()
                    val resp = client.newCall(req).execute()
                    val respJson = org.json.JSONObject(resp.body?.string() ?: "{}")
                    val reply = respJson.optJSONArray("content")
                        ?.optJSONObject(0)?.optString("text", "…") ?: "…"

                    // Check if AI is issuing a command
                    if (reply.startsWith("CMD:")) {
                        val parts = reply.removePrefix("CMD:").split(":", limit = 2)
                        if (parts.size == 2) {
                            val (nodeId, cmd) = parts
                            sendCommandSmart(nodeId.trim(), cmd.trim())
                            withContext(Dispatchers.Main) {
                                messages.add("assistant" to "✓ Sent $cmd to ${nodeId.trim()}")
                            }
                        }
                    } else {
                        withContext(Dispatchers.Main) {
                            messages.add("assistant" to reply)
                        }
                    }
                } catch (e: Exception) {
                    withContext(Dispatchers.Main) {
                        messages.add("assistant" to "Error: ${e.message}")
                    }
                } finally {
                    withContext(Dispatchers.Main) { isThinking = false }
                }
                // Scroll to bottom
                withContext(Dispatchers.Main) {
                    if (messages.isNotEmpty()) listState.animateScrollToItem(messages.size - 1)
                }
            }
        }

        Column(modifier = Modifier.fillMaxSize()) {

            // Header
            androidx.compose.foundation.layout.Row(
                modifier = Modifier
                    .fillMaxWidth()
                    .padding(horizontal = 12.dp, vertical = 8.dp),
                verticalAlignment = androidx.compose.ui.Alignment.CenterVertically
            ) {
                androidx.compose.material3.Text(
                    text = "TRACKIT AI",
                    color = androidx.compose.ui.graphics.Color.White,
                    style = androidx.compose.material3.MaterialTheme.typography.titleMedium,
                    modifier = Modifier.weight(1f)
                )
                // API key setup button
                androidx.compose.material3.TextButton(onClick = {
                    val prefs = getSharedPreferences(PREFS, Context.MODE_PRIVATE)
                    val input = android.widget.EditText(this@MainActivity).apply {
                        hint = "sk-ant-..."
                        setText(prefs.getString("anthropic_key", ""))
                        inputType = android.text.InputType.TYPE_CLASS_TEXT or
                                android.text.InputType.TYPE_TEXT_VARIATION_PASSWORD
                    }
                    AlertDialog.Builder(this@MainActivity)
                        .setTitle("Anthropic API Key")
                        .setMessage("Required for AI chat. Key stored locally.")
                        .setView(input)
                        .setPositiveButton("SAVE") { _, _ ->
                            prefs.edit().putString("anthropic_key", input.text.toString().trim()).apply()
                            showToast("API key saved")
                        }
                        .setNegativeButton("CANCEL", null)
                        .show()
                }) {
                    androidx.compose.material3.Text(
                        "API KEY",
                        style = androidx.compose.material3.MaterialTheme.typography.labelSmall,
                        color = androidx.compose.ui.graphics.Color.Gray
                    )
                }
            }

            // Message list
            androidx.compose.foundation.lazy.LazyColumn(
                state = listState,
                modifier = Modifier
                    .weight(1f)
                    .padding(horizontal = 10.dp),
                verticalArrangement = androidx.compose.foundation.layout.Arrangement.spacedBy(6.dp)
            ) {
                if (messages.isEmpty()) {
                    item {
                        androidx.compose.foundation.layout.Box(
                            modifier = Modifier.fillParentMaxSize(),
                            contentAlignment = androidx.compose.ui.Alignment.Center
                        ) {
                            Column(horizontalAlignment = androidx.compose.ui.Alignment.CenterHorizontally) {
                                androidx.compose.material3.Text(
                                    "Ask about nodes, detections,\nor send commands to devices.",
                                    color = androidx.compose.ui.graphics.Color.Gray,
                                    style = androidx.compose.material3.MaterialTheme.typography.bodySmall,
                                    textAlign = androidx.compose.ui.text.style.TextAlign.Center as androidx.compose.ui.text.style.TextAlign?
                                )
                            }
                        }
                    }
                }
                lazyItems(messages) { msg: Pair<String, String> ->
                    val (role, text) = msg
                    val isUser = role == "user"
                    androidx.compose.foundation.layout.Row(
                        modifier = Modifier.fillMaxWidth(),
                        horizontalArrangement = if (isUser)
                            androidx.compose.foundation.layout.Arrangement.End
                        else
                            androidx.compose.foundation.layout.Arrangement.Start
                    ) {
                        androidx.compose.material3.Surface(
                            shape = androidx.compose.foundation.shape.RoundedCornerShape(12.dp),
                            color = if (isUser)
                                androidx.compose.ui.graphics.Color(0xFF1565C0)
                            else
                                androidx.compose.ui.graphics.Color(0xFF2A2A2A),
                            modifier = Modifier.widthIn(max = 280.dp)
                        ) {

                                text = text,
                                color = androidx.compose.ui.graphics.Color.White,
                                style = androidx.compose.material3.MaterialTheme.typography.bodySmall,
                                modifier = Modifier.padding(horizontal = 12.dp, vertical = 8.dp)
                            )
                        }
                    }
                }
                if (isThinking) {
                    item {
                        androidx.compose.foundation.layout.Row {
                            androidx.compose.material3.Surface(
                                shape = androidx.compose.foundation.shape.RoundedCornerShape(12.dp),
                                color = androidx.compose.ui.graphics.Color(0xFF2A2A2A)
                            ) {
                                androidx.compose.material3.Text(
                                    "thinking…",
                                    color = androidx.compose.ui.graphics.Color.Gray,
                                    style = androidx.compose.material3.MaterialTheme.typography.bodySmall,
                                    modifier = Modifier.padding(horizontal = 12.dp, vertical = 8.dp)
                                )
                            }
                        }
                    }
                }
            }

                        .fillMaxWidth()
                        .padding(horizontal = 10.dp, vertical = 4.dp),
                    horizontalArrangement = androidx.compose.foundation.layout.Arrangement.spacedBy(6.dp)

                    modifier = Modifier.weight(1f),

                        androidx.compose.material3.Text(
                            "Message TrackIt AI…",
                            style = androidx.compose.material3.MaterialTheme.typography.bodySmall

                    },
                    singleLine = true,
                    keyboardOptions = androidx.compose.foundation.text.KeyboardOptions(
                        imeAction = androidx.compose.ui.text.input.ImeAction.Send
                    ),
                    keyboardActions = androidx.compose.foundation.text.KeyboardActions(
                        onSend = { sendMessage(inputText) }
